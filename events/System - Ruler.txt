# "Death of [Root.Monarch.GetTitle] [Root.Monarch.GetName]"
country_event = {
	id = fallout_ruler.1
	title = fallout_ruler.1.title
	desc = fallout_ruler.1.desc
    picture = USA_FLAG_eventPicture
	
	trigger = {
        has_regency = no
		NOT = { has_government_attribute = republic }
        NOT = { has_immortal_ruler = yes }
        NOT = { has_ruler_modifier = autodoc }
	}
	
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - fallout_ruler.1" }
        }
    }
    
	mean_time_to_happen = {
		months = 2400
		
        modifier = {
            factor = 0.9
            has_ruler_modifier = ruler_drug_abuse
        }
        
		modifier = {
			factor = 0.9
			is_monarch_leader = yes
		}
        
        modifier = {
			factor = 1.1
			is_female = yes
		}
        
		modifier = {
			factor = 2
			technology_group = tech_ghoul
		}
		modifier = {
			factor = 0.5
			technology_group = tech_super_mutant
		}
		
        modifier = {
			factor = 0.9
			ruler_age = 80
		}
		modifier = {
			factor = 0.9
			ruler_age = 90
		}
		modifier = {
			factor = 0.9
			ruler_age = 100
		}
	}

	option = {
		name = fallout_ruler.1.option.a

        kill_ruler = yes
	}
}

# "[Root.Monarch.GetName] changes!"
country_event = {
	id = fallout_ruler.3
	title = fallout_ruler.3.title
	desc = fallout_ruler.3.desc
    picture = USA_FLAG_eventPicture
	
	trigger = {
        has_regency = no
        has_immortal_ruler = yes
	}
	
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - fallout_ruler.3" }
        }
    }
    
	mean_time_to_happen = {
		months = 600
	}

	option = {
		name = fallout_ruler.3.option.a
		
		random_list = {
			50 = {
                if = { 
					limit = { adm = 6 }
					add_stability = 1
				}
				if = { 
					limit = { NOT = { adm = 6 } }
					change_adm = 1
				}
			}
            50 = {
                if = { 
					limit = { NOT = { adm = 2 } }
					add_stability = -1
				}
				if = { 
					limit = { adm = 2 }

				}
			}
