# "NCR Affirms Commitment"
country_event = {
	id = baja.1
	title = baja.1.title
	desc = baja.1.desc
    picture = NCR_RANGER_CAMP_eventPicture
	
    is_triggered_only = yes
	
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - baja.1" }
        }
    }
    
	option = {
		name = baja.1.option.a
		if = { 
		    limit = { 
			    80 = {
			        is_empty = yes
				}
			}
			80 = {
                cede_province = BAJ
			}
			add_core = 80
		}
		if = { 
		    limit = { 
			    79 = {
			        is_empty = yes
				}
			}
			79 = {
                cede_province = BAJ
            }
			add_core = 79
		}
		add_mil_power = 25
	}
}

# "Abandoned to the Ghosts"
country_event = {
	id = baja.2
	title = baja.2.title
	desc = baja.2.desc
    picture = RAIDERS_eventPicture
	
    is_triggered_only = yes
	
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - baja.1" }
        }
    }
    
	option = {
		name = baja.2.option.a
		if = { 
		    limit = { 
			    owns = 81
			}
			81 = {
			    change_culture = raider
			    change_religion = survivalist
			}
		}
        capital_scope = {
            spawn_rebels = {
				type = raider_rebels 
				size = 4
				win = yes
				unrest = 15
			}
        }
		ai_chance = {
			factor = 2
		}  
	}
	
	option = {
		name = baja.2.option.b
		if = { 
		    limit = { 
			    owns = 81
			}
			81 = {
			    change_culture = raider
			    change_religion = survivalist
			}
		}
		change_primary_culture = raider
		change_religion = survivalist
        
        if = {
            limit = { NOT = { has_reform = raider_reform } }
            change_government_effect = {
                TYPE = tribal
                REFORM = raider_reform
                LEGACY = raider_legacy
            }
        }
        
		ai_chance = {
			factor = 98
		}  
	}
}
