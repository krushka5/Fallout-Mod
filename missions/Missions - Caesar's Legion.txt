		}
	}
    
    # Home of the Legion
    fallout_home_of_the_legion = {
		icon = mission_usa_flag
        position = 2
        
		required_missions = {

            arizona_region = {
                type = all
                
                country_or_subject_holds = ROOT
            }
		}
		provinces_to_highlight = {
            region = arizona_region
			NOT = { country_or_subject_holds = ROOT }
		}
        
		effect = {

    # Assault on New Vegas
    fallout_assault_on_new_vegas = {
		icon = mission_usa_flag
        position = 3
        
		required_missions = { 
            fallout_home_of_the_legion
        }
        
		trigger = {
            if = {
                limit = {
                    exists = NVG
                }
                war_with = NVG
            }
            else = {
                new_vegas_area = {
                    type = all
                    country_or_subject_holds = ROOT
                }
            }
		}
		provinces_to_highlight = {
            if = {
                limit = {
                    exists = NVG

                    owned_by = NVG
                }

            else = {
