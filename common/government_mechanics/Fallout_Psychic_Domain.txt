psychic_domain_mechanic = {
	alert_icon_gfx = GFX_alerticons_government_mechanics
	alert_icon_index = 1
	available = {
		has_dlc = "Cradle of Civilization"
	}

	powers = {
		psychic_domain = {
			max = 100
			reset_on_new_ruler = yes
			base_monthly_growth = 1.0
		}
	}
	
	interactions = {
		psychic_domain_promote_open_discussion = {
			icon = GFX_psychic_domain_ability_1
			cost_type = psychic_domain
			cost = 50
            
			cooldown_years = 10
            
			trigger = {
                NOT = { has_country_modifier = psychic_domain_open_discussion }
			}
            
			effect = {
				add_country_modifier = {
                    name = psychic_domain_open_discussion
                    duration = 3650
                }
			}
            
			ai_chance = {
				factor = 10
			}
		}
		
        psychic_domain_establish_tribal_connections = {
			icon = GFX_psychic_domain_ability_2
			cost_type = psychic_domain
			cost = 30
            
			cooldown_years = 5
            
			trigger = {
                # Any friendly neighbor
                any_known_country = {
                    is_neighbor_of = ROOT
                    has_opinion = {
                        who = ROOT
                        value = 0
                    }
                }
			}
            
			effect = {
                if = {
                    limit = {
                        NOT = {
                            any_known_country = {
                                is_neighbor_of = ROOT
                                has_opinion = {
                                    who = ROOT
                                    value = 0
                                }
                            }
                        }
                    }
                    custom_tooltip = GAIN_OPINION_BOOST_IN_FRIENDLY_NEIGHBOR_TT
                }
                
                # Up to 3 random known friendly neighbors gain an opinion boost
				random_known_country = {
                    limit = {
                        is_neighbor_of = ROOT
                        has_opinion = {
                            who = ROOT
                            value = 0
                        }
                    }
                    type = random
                    amount = 3
                    
                    add_opinion = {
                        who = ROOT
                        modifier = psychic_domain_tribal_connections
                    }
                }
			}
            
			ai_chance = {
				factor = 10
			}
		}
        
		psychic_domain_prepare_for_war = {
			icon = GFX_psychic_domain_ability_3
			cost_type = psychic_domain
			cost = 50
            
			cooldown_years = 5
            
			trigger = {
				capital_scope = {
                    controlled_by = ROOT
                }
			}
            
			effect = {
				capital_scope = {
					add_unit_construction = {
						type = infantry
						amount = 5
						speed = 0.0
						cost = 0
					}
				}
			}
            
			ai_chance = {
				factor = 10
			}
		}
	}
}
