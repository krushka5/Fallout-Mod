unlock_schematic = {
    custom_tooltip = UNLOCK_schematic_$schematic$
    
    hidden_effect = {
        set_country_flag = schematic_$schematic$
        set_global_flag = discovered_schematic_$schematic$
    }
}

remove_schematic = {
    hidden_effect = {
        clr_country_flag = schematic_$schematic$
        clr_global_flag = discovered_schematic_$schematic$
    }
}

steal_schematic = {
    custom_tooltip = STEAL_schematic_$schematic$
    
    hidden_effect = {
        set_country_flag = schematic_$schematic$
        
        event_target:schematic_heist_target = {
            set_country_flag = schematic_$schematic$_stolen
            clr_country_flag = schematic_$schematic$
        }
    }
}

transfer_schematic = {
    hidden_effect = {
        if = {
            limit = {
                FROM = {
                    has_country_flag = schematic_$schematic$
                }
            }
            set_country_flag = schematic_$schematic$
            
            FROM = {
                clr_country_flag = schematic_$schematic$
            }
        }
    }
}

save_schematic_location = {
    if = {
        limit = {
            has_global_flag = discovered_schematic_$schematic$
        }
        every_country = {
            limit = {
                 has_country_flag = schematic_$schematic$
            }
            save_event_target_as = current_owner_$schematic$
        }
    }
    else = {
        REB = {
            save_event_target_as = current_owner_$schematic$
        }
    }
}

schematic_focus_cost = {
    # Primitive
    if = {
        limit = {
            is_primitive_scavenger = yes
        }
        
        add_treasury = -500
    }
    # Urban State
    else_if = {
        limit = {
            is_urban_scavenger = yes
        }
        
        add_treasury = -300
    }
    # Scientific State
    else = {
        add_treasury = -150
    }
    
    custom_tooltip = SCHEMATIC_FOCUS_EFFECT
}
