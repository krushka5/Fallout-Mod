#Define estate loyalty and influence modifiers here

estate_bureaucrats = {
	modifier_definition = {
		type = loyalty
		key = bureaucrats_loyalty_modifier
		trigger = {
			has_estate = estate_bureaucrats
		}
	}
	modifier_definition = {
		type = influence
		key = bureaucrats_influence_modifier
		trigger = {
			has_estate = estate_bureaucrats
		}
	}
    modifier_definition = {
		type = privileges
		key = bureaucrats_privilege_slots
		trigger = {
			has_estate = estate_bureaucrats
		}
	}
}
