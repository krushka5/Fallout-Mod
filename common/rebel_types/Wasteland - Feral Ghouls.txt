#------------------------------------------------------
# Feral Ghouls
#------------------------------------------------------
feral_ghoul_rebels = {
    color = { 60 60 60 }
    
	area = nation 					# Stays within the nation
	government = any			
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = yes
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = no
	
	artillery = 0.0
	infantry = 0.8
	cavalry = 0.2
	
	morale = 0.9
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = yes
	handle_action_build_core = yes	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 10
			culture_group = ghoul_group
            owner = {
        		NOT = { culture_group = ghoul_group }
         	}
		}
		modifier = {
			factor = 2
			religion = bright_brotherhood # Large population of sentient ghouls attract ferals
		}
		modifier = {
			factor = 1.2
			religion = renewal_cult # Large population of sentient ghouls attract ferals
		}
        modifier = {
			factor = 0.25
			culture_group = ghoul_group
           	owner = {
				culture_group = ghoul_group
       		}
		}
        
		modifier = {
			factor = 0
			NOT = {
				culture_group = ghoul_group
				owner = { religion = bright_brotherhood }
				owner = { religion = renewal_cult }
				religion = bright_brotherhood
				religion = renewal_cult
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 0.6
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
		modifier = {
			factor = 1.1
			NOT = {
				religion = bright_brotherhood
				religion = renewal_cult
			}
			owner = {
				OR = {
					religion = renewal_cult
					religion = bright_brotherhood
				}
			}
		}
		modifier = {
			factor = 2
			OR = {
				religion = bright_brotherhood
				religion = renewal_cult
			}
			owner = {
				NOT = {
					religion = renewal_cult
					religion = bright_brotherhood
				}
			}
		}
		modifier = {
			factor = 10
			culture_group = ghoul_group
			owner = {
				NOT = { culture_group = ghoul_group }
			}
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	
	siege_won_effect = {
		add_province_modifier = {
			name = damaged_population
			duration = 3650
		}
        add_devastation = 20
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "feral_ghoul_rebels_demand"
	
 	# Country Scope
	demands_enforced_effect = {
		add_prestige = -50
        add_stability = -2
        
        add_country_modifier = {
            name = damaged_population_global
            duration = 3650
        }
	}
}
