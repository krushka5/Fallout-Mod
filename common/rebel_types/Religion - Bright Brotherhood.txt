rebels_bright_brotherhood = {

	area = nation # Stays within the nation
	government = theocracy
	religion = bright_brotherhood
	defection = religion				# Prefers its religion
	independence = none			# Does not strive for independence
	unit_transfer = yes
	gfx_type = ghoul_gfx	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.0
	infantry = 0.8
	cavalry = 0.2
	
	morale = 0.9
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes	
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 2
			culture_group = ghoul_group
            owner = {
                NOT = { culture_group = ghoul_group }
            }
		}
		modifier = {
			factor = 2000
			has_missionary = yes
		}
        		modifier = {
			factor = 2
			culture_group = ghoul_group
           			owner = {
                			culture_group = ghoul_group
       			}
		}
		modifier = {
			factor = 0
			owner = { tag = GHH }
		}
		modifier = {
			factor = 0
			OR = {
				culture = hater
				owner = { primary_culture = hater }
			}
		}
		modifier = {
			factor = 0
			NOT = {
				culture = culture_ghoul
				religion = bright_brotherhood
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
		modifier = {
			factor = 10
			culture_group = ghoul_group
			owner = {
				NOT = { culture_group = ghoul_group }
		            }
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	
	siege_won_effect = {
		if = { limit = { has_global_flag = enable_common_logging } log = "rebels_bright_brotherhood.txt - siege_won_effect - [Root.GetName]" }
		custom_tooltip = religious_rebels_seige_effect_tooltip
		hidden_effect = {
			if = {
				limit = {
					OR = {
						religion = bright_brotherhood
						is_reformation_center = yes
						owner = { religion = bright_brotherhood }
					}
				}
				add_province_modifier = {
					name = "religious_violence"
					duration = 730
				}
			}
			if = {
				limit = {
					NOT = { religion = bright_brotherhood }
					is_reformation_center = no
					NOT = { owner = { religion = bright_brotherhood } }
				}
				change_religion = bright_brotherhood
				add_province_modifier = {
					name = "forced_conversion"
					duration = 730
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		OR = {
			religion_group = religion_group_pagan
			religion_group = religion_group_atomic
			culture_group = ghoul_group
			dominant_religion = bright_brotherhood
		}
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = ""
	
	# Country Scope	
	demands_enforced_effect = {
		if = { limit = { has_global_flag = enable_common_logging } log = "rebels_bright_brotherhood.txt - demands_enforced_effect - [Root.GetName]" }
		if = {
			limit = {
				NOT = {
					religion_group = religion_group_pagan
					religion_group = religion_group_atomic
				}
			}
			add_prestige = -100
			add_accepted_culture = ghoul
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
				}
				add_local_autonomy = 50
			}
		}
		if = {
			limit = {
				NOT = { religion = bright_brotherhood }
				OR = {
					dominant_religion = bright_brotherhood
					religion_group = religion_group_pagan
					religion_group = religion_group_atomic
					culture_group = ghoul_group
				}
			}
			change_religion = bright_brotherhood
			force_converted = yes
		}
		add_accepted_culture = ghoul
        		add_stability = -3
	}
}
